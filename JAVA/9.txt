class Solution {
    public boolean isPalindrome(int x) {
        int start = x;
        int n = x;
        int ans = 0;
        int temp = 0;
        while(n!=0){
        temp = n % 10;
        ans = ans*10 + temp;
        n = n/10;
        }
        System.out.printf("%d%d",ans,start);
        if(ans == start && start>=0){
            return true;
        }else{
            return false;
        }
    }
}
